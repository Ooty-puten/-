// Load all handlers dynamically
const handlers = {};
fs.readdirSync('./handlers').forEach(file => {
    if (file.endsWith('.js')) {
        const handlerName = file.slice(0, -3);
        handlers[handlerName] = require(`./handlers/${file}`);
    }
});

// Register commands
bot.onText(/\/start/, (msg) => handlers.start(bot, msg));
bot.onText(/\/help/, (msg) => handlers.help(bot, msg));
bot.onText(/\/contact/, (msg) => handlers.contact(bot, msg));

bot.onText(/\/register/, (msg) => handlers.register(bot, msg));

    if (!msg.text.startsWith('/')) {
        const response = getResponse(responseDatabase, msg.text);
        bot.sendMessage(msg.chat.id, response);
    }
});

games.js
const games = {
    rockPaperScissors: (bot, msg, choice) => {
        const chatId = msg.chat.id;
        const choices = ['Камень', 'Ножницы', 'Бумага'];
        const botChoice = choices[Math.floor(Math.random() * choices.length)];
        let result;

        if (choice === botChoice) {
            result = 'Ничья!';
        } else if (
            (choice === 'Камень' && botChoice === 'Ножницы') ||
            (choice === 'Ножницы' && botChoice === 'Бумага') ||
            (choice === 'Бумага' && botChoice === 'Камень')
        ) {
            result = 'Вы выиграли!';
        } else {
            result = 'Вы проиграли!';
        }

        bot.sendMessage(chatId, `Вы выбрали: ${choice}, я выбрал: ${botChoice}. ${result}`);
    },
    shooterGame: (bot, msg, opponentId) => {
        const chatId = msg.chat.id;
        const players = {};
        let currentTurn = msg.from.id;

        if (!opponentId) {
            bot.sendMessage(chatId, 'Пожалуйста, укажите ID противника.');
            return;
        }

        const getPlayer = (id) => {
            if (!players[id]) {
                players[id] = { aim: false };
            }
            return players[id];
        };

        const nextTurn = () => {
            currentTurn = currentTurn === msg.from.id ? opponentId : msg.from.id;
            bot.sendMessage(chatId, `Сейчас ходит: ${currentTurn === msg.from.id ? 'Вы' : 'Ваш противник'}`);
        };

        const endGame = (winnerId) => {
            bot.sendMessage(chatId, `Игра окончена. Победил: ${winnerId === msg.from.id ? 'Вы' : 'Ваш противник'}`);
            players[msg.from.id] = players[opponentId] = null;
        };

        const handleAction = (action, playerId) => {
            const player = getPlayer(playerId);
            if (action === 'прицелиться') {
                player.aim = true;
                bot.sendMessage(chatId, 'Вы прицелились. Ваш шанс на победу увеличен!');
            } else if (action === 'удар') {
                const opponent = getPlayer(playerId === msg.from.id ? opponentId : msg.from.id);
                opponent.aim = false;
                bot.sendMessage(chatId, 'Вы сбили прицел противника!');
            } else if (action === 'выстрел') {
                if (player.aim) {
                    endGame(playerId);
                } else {
                    bot.sendMessage(chatId, 'Вы промахнулись. Игра продолжается.');
                }
            } else {
                bot.sendMessage(chatId, 'Недопустимая команда. Пожалуйста, используйте: прицелиться, удар, выстрел.');
            }
            nextTurn();
        };

        bot.sendMessage(chatId, `Игра началась! Ваш противник: ${opponentId}. Ваш ход.`);
        bot.on('message', (response) => {
            if (response.chat.id !== chatId || response.from.id !== currentTurn) return;
            const action = response.text.toLowerCase();
            handleAction(action, response.from.id);
        });
    }
};

module.exports = {
    handleGame: (bot, msg, gameName, ...args) => {
        if (games[gameName]) {
            games[gameName](bot, msg, ...args);
        } else {
            bot.sendMessage(msg.chat.id, 'Игра не найдена. Проверьте доступные игры.');
        }
    }
};
